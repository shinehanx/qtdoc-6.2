   set(INSTALL_EXAMPLESDIR "examples")
 endif()

 find_package(Qt6 COMPONENTS Core)

 find_package(Qt6 COMPONENTS QmlXmlListModel)

     main.cpp
 )

     WIN32_EXECUTABLE TRUE
     MACOSX_BUNDLE TRUE
 )

     Qt::Core
     Qt::Gui
     Qt::Qml
     Qt::Quick
     Qt::QmlXmlListModel
 )
